地标动作

warp_create​

action:
  - "[warp_create]"

warp_delete​

action:
  - "[warp_delete] 地标名称"
  - "[warp_delete]"  # 删除当前选中的地标

warp_info​

action:
  - "[warp_info] 地标名称"
  - "[warp_info]"  # 显示当前选中地标信息

action:
